```kotlin
override fun rsaEncrypt(data: ByteArray, publicKeyPem: String, padding: String): ByteArray {
    val clean = publicKeyPem.lines().filterNot { it.startsWith("-----") }.joinToString("").trim()
    val keySpec = X509EncodedKeySpec(Base64.getDecoder().decode(clean))
    val key = KeyFactory.getInstance("RSA").generatePublic(keySpec)
    val transform = when (padding.uppercase()) {
        "NOPADDING", "NONE" -> "RSA/ECB/NoPadding"
        "PKCS1" -> "RSA/ECB/PKCS1Padding"
        else -> "RSA/ECB/${padding}Padding"
    }
    val cipher = Cipher.getInstance(transform)
    cipher.init(Cipher.ENCRYPT_MODE, key)
    return cipher.doFinal(data)
}

override fun zlibInflate(input: ByteArray, format: String): ByteArray {
    val detected = when (format.lowercase()) {
        "auto" -> detectZlibFormat(input)
        else -> format.lowercase()
    }
    return when (detected) {
        "gzip" -> {
            java.util.zip.GZIPInputStream(input.inputStream()).use { it.readAllBytes() }
        }
        "raw" -> {
            val inflater = java.util.zip.Inflater(true)  // nowrap=true 即 raw deflate
            inflater.setInput(input)
            val out = ByteArray(input.size * 16)
            val n = inflater.inflate(out)
            inflater.end()
            out.copyOf(n)
        }
        else -> {  // zlib
            val inflater = java.util.zip.Inflater()
            inflater.setInput(input)
            val out = ByteArray(input.size * 16)
            val n = inflater.inflate(out)
            inflater.end()
            out.copyOf(n)
        }
    }
}

private fun detectZlibFormat(bytes: ByteArray): String {
    if (bytes.size < 2) return "raw"
    val b0 = bytes[0].toInt() and 0xFF
    val b1 = bytes[1].toInt() and 0xFF
    return when {
        b0 == 0x1F && b1 == 0x8B -> "gzip"
        b0 == 0x78 -> "zlib"   // 0x78 0x01/0x9C/0xDA 是常见 zlib header
        else -> "raw"
    }
}
```

```kotlin
interface JsRuntime : AutoCloseable {
    suspend fun evaluate(script: String, name: String = "anon.js"): JsValue
    suspend fun invoke(path: String, vararg args: JsValue): JsValue

    /**
     * 注册 suspend host 函数到 JS 全局作用域。
     *
     * M1 行为：JS 调用该函数时立即拿到一个 Promise；Kotlin 后台在 dispatcher 协程
     * scope 上执行 host(args) 并 resolve/reject Promise。JS 侧应 await 或 then。
     * 适用于 IO / 网络 / 真正耗时的 host 函数。
     */
    fun register(name: String, host: HostFunction)

    /**
     * 注册纯同步 host 函数到 JS 全局作用域。
     *
     * JS 调用立即返回 host 结果，不经 Promise。host 内不得调用 suspend 函数或阻塞 IO。
     * 适用于 md5/aes/base64/bufferFrom/zlib/iconv/log/platformTag 等 compute-only bridge。
     */
    fun registerSync(name: String, host: (List<JsValue>) -> JsValue)

    val dispatcher: CoroutineDispatcher
    val sourceId: String
}

expect object JsRuntimeFactory {
    fun create(sourceId: String, bridge: JsBridge): JsRuntime
}
```

```kotlin
    .allowHostAccess(HostAccess.ALL)
    .option("js.top-level-await", "true")
    .option("js.ecmascript-version", "2022")

private val hostScope = kotlinx.coroutines.CoroutineScope(

override fun register(name: String, host: HostFunction) {
    runBlocking(dispatcher) {
        withContext0 {
            val promiseCtor = context.getBindings("js").getMember("Promise")
            val proxy = ProxyExecutable { graalArgs ->
                // graalArgs 是 ProxyExecutable 传入的 JS 参数
                val parsed = graalArgs.map { it.toJsValue() }

                val executor = ProxyExecutable { ctorArgs ->
                    resolveFn = ctorArgs[0]
                    rejectFn = ctorArgs[1]
                    null

                        // 切回 JS 线程调 resolve

                            }

                            withContext0 {
                                rejectFn!!.execute(t.message ?: t::class.simpleName ?: "error")
                            }
                        }
                    }
                }

                promise  // 同步返回 Promise
            }
            context.getBindings("js").putMember(name, proxy)
        }
    }
}
```

```kotlin
import kotlinx.cinterop.*
import kotlinx.coroutines.*
import platform.Foundation.NSArray
import platform.Foundation.NSData
import platform.Foundation.NSObject
import platform.Foundation.NSNull
import platform.Foundation.NSNumber
import platform.Foundation.NSString
import platform.Foundation.numberWithBool
import platform.Foundation.numberWithDouble
import platform.JavaScriptCore.JSContext
import platform.JavaScriptCore.JSValue as NativeJSValue

@OptIn(ExperimentalForeignApi::class)
actual object JsRuntimeFactory {
    actual fun create(sourceId: String, bridge: JsBridge): JsRuntime = JsCoreRuntime(sourceId, bridge)
}

@OptIn(ExperimentalForeignApi::class, DelicateCoroutinesApi::class)
private class JsCoreRuntime(
    override val sourceId: String,
    private val bridge: JsBridge,
) : JsRuntime {

    override val dispatcher: CoroutineDispatcher = newSingleThreadContext("jsc-$sourceId")
    private val ctx: JSContext = JSContext()
    private val hostScope = CoroutineScope(SupervisorJob() + dispatcher)

    init {
        runBlocking(dispatcher) {
            ctx.evaluateScript("globalThis.lyn = globalThis.lyn || {};")
        }
    }

    override suspend fun evaluate(script: String, name: String): JsValue = withContext(dispatcher) {
        try {
            val v = ctx.evaluateScript(script) ?: return@withContext JsValue.Undefined
            if (ctx.exception != null) {
                val exc = ctx.exception!!
                val msg = exc.toString() ?: "js error"
                ctx.exception = null
                throw MusicSourceException.ScriptRuntimeError(sourceId, msg, null)
            }
            v.toJsValue()
        } catch (t: Throwable) {
            if (t is MusicSourceException) throw t
            throw MusicSourceException.ScriptRuntimeError(sourceId, t.message ?: "", t)
        }
    }

    override suspend fun invoke(path: String, vararg args: JsValue): JsValue = withContext(dispatcher) {
        try {
            var cur: NativeJSValue? = ctx.globalObject
            for (seg in path.split('.')) {
                cur = cur?.objectForKeyedSubscript(seg)
                if (cur == null || cur.isUndefined()) {
                    throw MusicSourceException.ScriptRuntimeError(sourceId, "path not found: $path", null)
                }
            }
            val nsArgs = args.map { it.toNSObject() }
            val result = cur!!.callWithArguments(nsArgs)
            if (ctx.exception != null) {
                val exc = ctx.exception!!
                val msg = exc.toString() ?: "js error"
                ctx.exception = null
                throw MusicSourceException.ScriptRuntimeError(sourceId, msg, null)
            }
            result?.toJsValue() ?: JsValue.Undefined
        } catch (t: Throwable) {
            if (t is MusicSourceException) throw t
            throw MusicSourceException.ScriptRuntimeError(sourceId, t.message ?: "", t)
        }
    }

    override fun register(name: String, host: HostFunction) {
        runBlocking(dispatcher) {
            // M1.0 Apple：同步占位实现（阻塞调 host）。真异步 Promise 在 M1.1 Task 4 做。
            // 由于 Apple JSC 的 Promise 创建需要 valueWithNewPromiseInContext + block 捕获 resolve/reject
            // 工程复杂且 iOS 13+ 才稳定，M1.0 暂 runBlocking 兼容 M0 行为
            val block: (NSArray?) -> NSObject? = { nsArgs ->
                val count = nsArgs?.count?.toInt() ?: 0
                val kArgs = (0 until count).map { i ->
                    (nsArgs!!.objectAtIndex(i.toULong()) as NativeJSValue).toJsValue()
                }
                runBlocking {
                    host(kArgs).toNSObject()
                }
            }
            val fn = NativeJSValue.valueWithObject(block as Any, ctx)
            ctx.globalObject!!.setObject(fn, forKeyedSubscript = name as NSObject)
        }
    }

    override fun registerSync(name: String, host: (List<JsValue>) -> JsValue) {
        runBlocking(dispatcher) {
            val block: (NSArray?) -> NSObject? = { nsArgs ->
                val count = nsArgs?.count?.toInt() ?: 0
                val kArgs = (0 until count).map { i ->
                    (nsArgs!!.objectAtIndex(i.toULong()) as NativeJSValue).toJsValue()
                }
                host(kArgs).toNSObject()
            }
            val fn = NativeJSValue.valueWithObject(block as Any, ctx)
            ctx.globalObject!!.setObject(fn, forKeyedSubscript = name as NSObject)
        }
    }

    override fun close() {
        hostScope.cancel()
        // JSC JSContext 由 GC 负责；显式清 global 防 retain cycle
        try {
            runBlocking(dispatcher) {
                ctx.evaluateScript("globalThis.lyn = null;")
            }
        } catch (_: Throwable) {}
        (dispatcher as? ExecutorCoroutineDispatcher)?.close()
    }
}

@OptIn(ExperimentalForeignApi::class)
private fun NativeJSValue.toJsValue(): JsValue = when {
    isUndefined() -> JsValue.Undefined
    isNull() -> JsValue.Null
    isBoolean() -> JsValue.Bool(toBool())
    isNumber() -> JsValue.Num(toDouble())
    isString() -> JsValue.Str(toString() ?: "")
    isArray() -> {
        val len = (valueForProperty("length")?.toDouble() ?: 0.0).toInt()
        JsValue.Arr((0 until len).map { i ->
            objectAtIndexedSubscript(i.toULong())?.toJsValue() ?: JsValue.Undefined
        })
    }
    isObject() -> {
        // 用 JSON.stringify 兜底遍历 object properties
        val ctx = this.context
        val stringifyFn = ctx?.evaluateScript("JSON.stringify")
        val json = stringifyFn?.callWithArguments(listOf(this))?.toString() ?: "null"
        jsonToJsValue(json)
    }
    else -> JsValue.Undefined
}

@OptIn(ExperimentalForeignApi::class)
private fun JsValue.toNSObject(): NSObject? = when (this) {
    is JsValue.Null -> NSNull()
    is JsValue.Undefined -> null
    is JsValue.Bool -> NSNumber.numberWithBool(value)
    is JsValue.Num -> NSNumber.numberWithDouble(value)
    is JsValue.Str -> value as NSString
    is JsValue.Bytes -> data.toNSData()
    is JsValue.Arr -> {
        val list = items.map { (it.toNSObject() ?: NSNull()) as Any }
        (list as List<Any>) as NSArray
    }
    is JsValue.Obj -> {
        // Dict 走 NSDictionary；这里简化用 JSON.stringify 逆向 — 需要 ctx 构造 JSValue parse
        // M1.0 mini 实现：对 Obj 只写 key/value 均为 primitive 的情形，复杂嵌套暂依赖 JSON
        platform.Foundation.NSMutableDictionary().apply {
            entries.forEach { (k, v) -> setObject(v.toNSObject() ?: NSNull(), forKey = k as NSString) }
        } as NSObject
    }
}

@OptIn(ExperimentalForeignApi::class)
private fun ByteArray.toNSData(): NSData = usePinned { pinned ->
    NSData.create(bytes = pinned.addressOf(0), length = this.size.toULong())
}

/** 简易 JSON → JsValue；依赖 kotlinx.serialization.json 已是 commonMain 依赖 */
private fun jsonToJsValue(json: String): JsValue {
    if (json == "null" || json.isEmpty()) return JsValue.Null
    return try {
        val element = kotlinx.serialization.json.Json.parseToJsonElement(json)
        element.toJsValue()
    } catch (_: Throwable) {
        JsValue.Str(json)
    }
}

private fun kotlinx.serialization.json.JsonElement.toJsValue(): JsValue = when (this) {
    is kotlinx.serialization.json.JsonNull -> JsValue.Null
    is kotlinx.serialization.json.JsonPrimitive -> when {
        this.isString -> JsValue.Str(this.content)
        this.booleanOrNull != null -> JsValue.Bool(this.boolean)
        else -> this.doubleOrNull?.let { JsValue.Num(it) } ?: JsValue.Str(this.content)
    }
    is kotlinx.serialization.json.JsonArray -> JsValue.Arr(this.map { it.toJsValue() })
    is kotlinx.serialization.json.JsonObject -> JsValue.Obj(this.mapValues { it.value.toJsValue() })
}
```

```kotlin
package top.iwesley.lyn.music.online.source

import kotlinx.cinterop.*
import platform.CommonCrypto.*
import platform.Foundation.*
import platform.Security.*
import platform.posix.memcpy

@OptIn(ExperimentalForeignApi::class)
actual fun createPlatformCrypto(): PlatformCrypto = DarwinPlatformCrypto()

@OptIn(ExperimentalForeignApi::class)
class DarwinPlatformCrypto : PlatformCrypto {

    // ============ hash ============
    override fun md5Hex(input: ByteArray): String {
        val out = UByteArray(CC_MD5_DIGEST_LENGTH)
        input.usePinned { pin ->
            out.usePinned { outPin ->
                CC_MD5(pin.addressOf(0), input.size.toUInt(), outPin.addressOf(0))
            }
        }
        return out.joinToString("") { it.toInt().toString(16).padStart(2, '0') }
    }

    override fun sha1(input: ByteArray): ByteArray {
        val out = UByteArray(CC_SHA1_DIGEST_LENGTH)
        input.usePinned { pin ->
            out.usePinned { outPin ->
                CC_SHA1(pin.addressOf(0), input.size.toUInt(), outPin.addressOf(0))
            }
        }
        return out.toByteArray()
    }

    override fun sha256(input: ByteArray): ByteArray {
        val out = UByteArray(CC_SHA256_DIGEST_LENGTH)
        input.usePinned { pin ->
            out.usePinned { outPin ->
                CC_SHA256(pin.addressOf(0), input.size.toUInt(), outPin.addressOf(0))
            }
        }
        return out.toByteArray()
    }

    // ============ AES ============
    override fun aesEncrypt(data: ByteArray, key: ByteArray, iv: ByteArray?, mode: String): ByteArray {
        val isEcb = mode.uppercase() == "ECB"
        val options = (kCCOptionPKCS7Padding or (if (isEcb) kCCOptionECBMode else 0u))
        return doCrypt(kCCEncrypt, kCCAlgorithmAES, options, key, iv, data)
    }

    override fun desEncrypt(data: ByteArray, key: ByteArray, iv: ByteArray?, mode: String): ByteArray =
        throw NotImplementedError("DES on Apple not implemented; no lx source uses DES as of bundle-M1.0")

    private fun doCrypt(
        op: CCOperation, algo: CCAlgorithm, options: CCOptions,
        key: ByteArray, iv: ByteArray?, data: ByteArray,
    ): ByteArray {
        val outSize = data.size + kCCBlockSizeAES128.toInt()
        val out = ByteArray(outSize)
        memScoped {
            val dataOutMoved = alloc<ULongVar>()
            val status = key.usePinned { keyPin ->
                data.usePinned { dataPin ->
                    out.usePinned { outPin ->
                        val ivPin = iv?.pin()
                        try {
                            CCCrypt(
                                op, algo, options,
                                keyPin.addressOf(0), key.size.toULong(),
                                ivPin?.addressOf(0),
                                dataPin.addressOf(0), data.size.toULong(),
                                outPin.addressOf(0), outSize.toULong(),
                                dataOutMoved.ptr
                            )
                        } finally {
                            ivPin?.unpin()
                        }
                    }
                }
            }
            if (status != kCCSuccess) {
                throw RuntimeException("CCCrypt failed: status=$status")
            }
            return out.copyOf(dataOutMoved.value.toInt())
        }
    }

    // ============ RSA ============
    override fun rsaEncrypt(data: ByteArray, publicKeyPem: String, padding: String): ByteArray {
        val derBody = publicKeyPem.lines().filterNot { it.startsWith("-----") }.joinToString("").trim()
        val spki = base64Decode(derBody)
        // X.509 SPKI 剥头：定长 24 字节 RSA 1024-bit/2048-bit 前缀；
        // 1024 bit 前缀字节(hex): 30 81 9F 30 0D 06 09 2A 86 48 86 F7 0D 01 01 01 05 00 03 81 8D 00 30 81 89
        // 2048 bit 前缀字节：30 82 01 22 30 0D 06 09 2A 86 48 86 F7 0D 01 01 01 05 00 03 82 01 0F 00 30 82 01 0A
        val pkcs1 = stripSpkiHeader(spki)

        val algo = when (padding.uppercase()) {
            "NOPADDING", "NONE", "RAW" -> kSecKeyAlgorithmRSAEncryptionRaw
            "PKCS1" -> kSecKeyAlgorithmRSAEncryptionPKCS1
            else -> kSecKeyAlgorithmRSAEncryptionPKCS1
        }

        memScoped {
            val attrs = mapOf(
                kSecAttrKeyType to kSecAttrKeyTypeRSA,
                kSecAttrKeyClass to kSecAttrKeyClassPublic,
            ).toCFDictionary()
            val err = alloc<CFErrorRefVar>()
            val keyData = pkcs1.toNSData()
            val secKey = SecKeyCreateWithData(keyData as CFDataRef, attrs, err.ptr)
                ?: throw RuntimeException("SecKeyCreateWithData failed")

            val cipherData = data.toNSData()
            val encrypted = SecKeyCreateEncryptedData(secKey, algo, cipherData as CFDataRef, err.ptr)
                ?: throw RuntimeException("SecKeyCreateEncryptedData failed")

            val nsData = encrypted as NSData
            return nsData.toByteArray()
        }
    }

    private fun stripSpkiHeader(spki: ByteArray): ByteArray {
        // 若 SPKI 格式（以 0x30 0x81 或 0x30 0x82 开头），定长剥头
        if (spki.size < 30) return spki
        return when {
            spki[0] == 0x30.toByte() && spki[1] == 0x81.toByte() -> {
                // 1024-bit: 25 字节前缀（含 BIT STRING 0x03 0x81 0x8D 0x00 + 内层 SEQUENCE header）
                spki.copyOfRange(25, spki.size)
            }
            spki[0] == 0x30.toByte() && spki[1] == 0x82.toByte() -> {
                // 2048-bit: 26 字节前缀
                spki.copyOfRange(26, spki.size)
            }
            else -> spki  // 已经是 PKCS#1 raw
        }
    }

    // ============ base64 / hex ============
    override fun base64Encode(input: ByteArray): String =
        input.toNSData().base64EncodedStringWithOptions(0u)

    override fun base64Decode(input: String): ByteArray =
        (NSData.create(base64EncodedString = input, options = 0u) ?: return ByteArray(0)).toByteArray()

    override fun hexEncode(input: ByteArray): String =
        input.joinToString("") { (it.toInt() and 0xFF).toString(16).padStart(2, '0') }

    override fun hexDecode(input: String): ByteArray =
        input.chunked(2).map { it.toInt(16).toByte() }.toByteArray()

    // ============ zlib ============
    override fun zlibInflate(input: ByteArray, format: String): ByteArray {
        val actual = when (format.lowercase()) {
            "auto" -> detectZlibFormat(input)
            else -> format.lowercase()
        }
        val windowBits = when (actual) {
            "gzip" -> 47    // 15 + 32 (gzip auto)
            "zlib" -> 15
            "raw" -> -15
            else -> 47
        }
        return inflateWithZlib(input, windowBits)
    }

    private fun detectZlibFormat(bytes: ByteArray): String {
        if (bytes.size < 2) return "raw"
        val b0 = bytes[0].toInt() and 0xFF
        val b1 = bytes[1].toInt() and 0xFF
        return when {
            b0 == 0x1F && b1 == 0x8B -> "gzip"
            b0 == 0x78 -> "zlib"
            else -> "raw"
        }
    }

    private fun inflateWithZlib(input: ByteArray, windowBits: Int): ByteArray {
        // 使用 platform.zlib.inflate 流式 API
        memScoped {
            val stream = alloc<platform.zlib.z_stream>()
            stream.zalloc = null
            stream.zfree = null
            stream.opaque = null
            val initResult = platform.zlib.inflateInit2(stream.ptr, windowBits)
            if (initResult != platform.zlib.Z_OK) {
                throw RuntimeException("inflateInit2 failed: $initResult")
            }
            try {
                return input.usePinned { inPin ->
                    stream.next_in = inPin.addressOf(0).reinterpret()
                    stream.avail_in = input.size.toUInt()
                    val outChunks = mutableListOf<ByteArray>()
                    while (true) {
                        val outBuf = ByteArray(16384)
                        outBuf.usePinned { outPin ->
                            stream.next_out = outPin.addressOf(0).reinterpret()
                            stream.avail_out = outBuf.size.toUInt()
                            val res = platform.zlib.inflate(stream.ptr, platform.zlib.Z_NO_FLUSH)
                            val used = outBuf.size - stream.avail_out.toInt()
                            if (used > 0) outChunks += outBuf.copyOf(used)
                            if (res == platform.zlib.Z_STREAM_END) return@usePinned
                            if (res != platform.zlib.Z_OK) throw RuntimeException("inflate error: $res")
                        }
                        if (stream.avail_in.toInt() == 0 && stream.avail_out.toInt() != 0) break
                    }
                    outChunks.fold(ByteArray(0)) { acc, c -> acc + c }
                }
            } finally {
                platform.zlib.inflateEnd(stream.ptr)
            }
        }
    }

    // ============ iconv ============
    override fun iconvDecode(input: ByteArray, encoding: String): String {
        val enc = appleEncoding(encoding)
        val ns = NSString.create(data = input.toNSData(), encoding = enc)
        return ns?.toString() ?: input.decodeToString()
    }

    override fun iconvEncode(input: String, encoding: String): ByteArray {
        val enc = appleEncoding(encoding)
        val ns = (input as NSString).dataUsingEncoding(enc) ?: return input.encodeToByteArray()
        return (ns as NSData).toByteArray()
    }

    private fun appleEncoding(name: String): NSStringEncoding = when (name.lowercase()) {
        "gbk", "gb18030", "gb2312" -> CFStringConvertEncodingToNSStringEncoding(kCFStringEncodingGB_18030_2000)
        "utf-8", "utf8" -> NSUTF8StringEncoding
        "ascii" -> NSASCIIStringEncoding
        else -> NSUTF8StringEncoding
    }
}

// ============ NSData ↔ ByteArray 辅助 ============
@OptIn(ExperimentalForeignApi::class)
private fun ByteArray.toNSData(): NSData =
    this.usePinned { NSData.create(bytes = it.addressOf(0), length = this.size.toULong()) }

@OptIn(ExperimentalForeignApi::class)
private fun NSData.toByteArray(): ByteArray {
    val size = this.length.toInt()
    if (size == 0) return ByteArray(0)
    val out = ByteArray(size)
    out.usePinned { pin -> memcpy(pin.addressOf(0), this.bytes, size.toULong()) }
    return out
}

@OptIn(ExperimentalForeignApi::class)
private fun Map<CFTypeRef?, CFTypeRef?>.toCFDictionary(): CFDictionaryRef {
    // 简化实现，使用 NSDictionary bridge
    val dict = NSMutableDictionary()
    this.forEach { (k, v) -> dict.setObject(v as Any, forKey = k as NSCopyingProtocol) }
    @Suppress("UNCHECKED_CAST")
    return dict as CFDictionaryRef
}
```

```kotlin
import top.iwesley.lyn.music.online.resolve.SourceUrlResolver

data class ResolvedUrl(val url: String, val quality: Quality, val sourceId: String)

interface SongUrlResolver {
    suspend fun resolve(id: OnlineMusicId, preferredQuality: Quality, songContext: OnlineSong? = null): ResolvedUrl
}

class DefaultSongUrlResolver(
    private val repository: OnlineMusicRepository,
    private val findMusic: FindMusicM0,
    private val sourceResolvers: Map<String, SourceUrlResolver> = emptyMap(),   // NEW
) : SongUrlResolver {

    override suspend fun resolve(
        id: OnlineMusicId,
        preferredQuality: Quality,
        songContext: OnlineSong?,
    ): ResolvedUrl {
        val degrade = Quality.degradationOrder.dropWhile { it != preferredQuality }
            .ifEmpty { Quality.degradationOrder }

        // 同源降级：优先用 Kotlin SourceUrlResolver，失败 fallback 到 JS (repository.getPlayableUrl)
        for (q in degrade) {
            val kotlinResolver = sourceResolvers[id.source]
            try {
                val url = if (kotlinResolver != null) {
                    kotlinResolver.resolve(id.songmid, q)
                } else {
                    repository.getPlayableUrl(id, q)
                }
                log(id.source, "url-ok q=${q.lxKey} via=${if (kotlinResolver != null) "kotlin" else "js"}")
                return ResolvedUrl(url.url, q, id.source)
            } catch (e: MusicSourceException) {
                log(id.source, "url-fail q=${q.lxKey}: ${e.message}", DiagnosticLogLevel.WARN, e)
                continue
            }
        }

        // 跨源 findMusic
        if (songContext != null) {
            val alt = findMusic.find(songContext, excludeSource = id.source)
            if (alt != null) {
                try {
                    val altResolver = sourceResolvers[alt.id.source]
                    val url = if (altResolver != null) {
                        altResolver.resolve(alt.id.songmid, alt.defaultQuality)
                    } else {
                        repository.getPlayableUrl(alt.id, alt.defaultQuality)
                    }
```

```kotlin
package top.iwesley.lyn.music.online.resolve

import io.ktor.client.HttpClient
import io.ktor.client.request.*
import io.ktor.client.statement.*
import io.ktor.http.*
import kotlinx.serialization.json.*
import top.iwesley.lyn.music.online.source.PlatformCrypto
import top.iwesley.lyn.music.online.types.PlayableUrl
import top.iwesley.lyn.music.online.types.Quality
import top.iwesley.lyn.music.scripting.MusicSourceException

/**
 * tx 源（QQ 音乐）URL 拿取器。
 *
 * 端点：https://u.y.qq.com/cgi-bin/musics.fcg
 * 签名：zzcSign（裸 SHA1(text) + pick/XOR/base64 scramble，见 vendor/lx-sdk/tx/utils/crypto.js:17-24）。无 salt。
 * 依赖：PlatformCrypto.sha1 / hexEncode / base64Encode；DeviceFingerprint.guid
 */
class TxUrlResolver(
    private val http: HttpClient,
    private val crypto: PlatformCrypto,
    private val fingerprint: DeviceFingerprint,
) : SourceUrlResolver {

    companion object {
        private const val ENDPOINT = "https://u.y.qq.com/cgi-bin/musics.fcg"
        private val QUALITY_TO_FORMAT = mapOf(
            "flac24bit" to "RS01",   // Hi-Res
            "flac" to "F000",
            "320k" to "M800",
            "192k" to "M500",        // AAC 192k
            "128k" to "M500",        // 降级到 M500
        )
    }

    override suspend fun resolve(songmid: String, quality: Quality): PlayableUrl {
        val format = QUALITY_TO_FORMAT[quality.lxKey]
            ?: throw MusicSourceException.SourceDisabled("tx", "quality ${quality.lxKey} unsupported")
        val filename = "${format}${songmid}.${if (format == "F000") "flac" else "mp3"}"

        val payload = buildJsonObject {
            put("comm", buildJsonObject {
                put("uin", "0")
                put("format", "json")
                put("ct", 19)
                put("cv", 1873)
                put("guid", fingerprint.guid)
            })
            put("req_0", buildJsonObject {
                put("module", "vkey.GetVkeyServer")
                put("method", "CgiGetVkey")
                put("param", buildJsonObject {
                    put("guid", fingerprint.guid)
                    put("songmid", buildJsonArray { add(songmid) })
                    put("filename", buildJsonArray { add(filename) })
                    put("songtype", buildJsonArray { add(0) })
                    put("uin", "0")
                    put("loginflag", 1)
                    put("platform", "20")
                })
            })
        }.toString()

        val sign = zzcSign(payload)

        val resp = http.post(ENDPOINT) {
            url.parameters.append("sign", sign)
            contentType(ContentType.Application.Json)
            setBody(payload)
        }
        val text = resp.bodyAsText()
        val parsed = Json.parseToJsonElement(text).jsonObject

        val req0 = parsed["req_0"]?.jsonObject
            ?: throw MusicSourceException.Parse("tx", null)
        val data = req0["data"]?.jsonObject
            ?: throw MusicSourceException.Parse("tx", null)
        val sip = data["sip"]?.jsonArray?.firstOrNull()?.jsonPrimitive?.content
            ?: throw MusicSourceException.Parse("tx", null)
        val info = data["midurlinfo"]?.jsonArray?.firstOrNull()?.jsonObject
            ?: throw MusicSourceException.Parse("tx", null)
        val purl = info["purl"]?.jsonPrimitive?.content
            ?: throw MusicSourceException.UrlExpired("tx")
        if (purl.isEmpty()) throw MusicSourceException.UrlExpired("tx")

        return PlayableUrl(
            url = sip + purl,
            quality = quality,
            fetchedAt = kotlin.time.Clock.System.now(),
        )
    }

    /**
     * lx zzcSign（见 vendor/lx-sdk/tx/utils/crypto.js:17-24）：
     *   hash = SHA1(text)                                         // hex string (40 chars)
     *   part1 = pick(hash, [23,14,6,36,16,40,7,19])                // 8 chars
     *   part2 = pick(hash, [16,1,32,12,19,27,8,5])                 // 8 chars
     *   part3 = scramble[i] XOR hash_bytes[i] for i in 0..19       // 20 bytes
     *   b64   = base64(part3).replace(/[\\/+=]/g, '')
     *   return "zzc${part1}${b64}${part2}".toLowerCase()
     * 注意：输入是裸 payload，**不**预先加 salt。SHA1 摘要用 hex 字符串做 pick（pick 的是 hex 字符，不是字节）。
     */
    private fun zzcSign(payload: String): String {
        val hashBytes = crypto.sha1(payload.encodeToByteArray())
        val hashHex = crypto.hexEncode(hashBytes)
        val part1 = PICK_PART_1.map { hashHex[it] }.joinToString("")
        val part2 = PICK_PART_2.map { hashHex[it] }.joinToString("")
        val part3 = ByteArray(SCRAMBLE.size) { i ->
            (SCRAMBLE[i] xor (hashBytes[i].toInt() and 0xFF)).toByte()
        }
        val b64 = crypto.base64Encode(part3).replace(Regex("[\\\\/+=]"), "")
        return "zzc$part1$b64$part2".lowercase()
    }

    companion object {
        private val PICK_PART_1 = intArrayOf(23, 14, 6, 36, 16, 40, 7, 19)
        private val PICK_PART_2 = intArrayOf(16, 1, 32, 12, 19, 27, 8, 5)
        private val SCRAMBLE = intArrayOf(89, 39, 179, 150, 218, 82, 58, 252, 177, 52, 186, 123, 120, 64, 242, 133, 143, 161, 121, 179)
    }
}
```

**注**：真算法已在本 plan 内写出，落地时直接用；**不需要**保留 M0 风格的 stub。若测试意外失败（如 `part1` index 40 超过 hex 长度 40，注意 hex 有 40 字符，index 40 即越界），使用 `hashHex.getOrNull(it) ?: '0'` 保护。lx JS 侧 `hash[40]` 在 SHA1 hex 场景会 undefined 被 join 成空字符串。

```kotlin
package top.iwesley.lyn.music.online.resolve

import io.ktor.client.HttpClient
import io.ktor.client.request.*
import io.ktor.client.statement.*
import io.ktor.http.*
import kotlinx.serialization.json.*
import top.iwesley.lyn.music.online.source.PlatformCrypto
import top.iwesley.lyn.music.online.types.PlayableUrl
import top.iwesley.lyn.music.online.types.Quality
import top.iwesley.lyn.music.scripting.MusicSourceException

/**
 * wy 源（网易云）URL 拿取器。
 *
 * 端点：http://interface3.music.163.com/eapi/song/enhance/player/url
 * 签名：eapi（AES-ECB(key="e82ckenh8dichen8") + MD5）
 */
class WyUrlResolver(
    private val http: HttpClient,
    private val crypto: PlatformCrypto,
) : SourceUrlResolver {

    companion object {
        private const val ENDPOINT = "http://interface3.music.163.com/eapi/song/enhance/player/url"
        private const val EAPI_KEY = "e82ckenh8dichen8"  // 固定
        private const val EAPI_URL_PATH = "/api/song/enhance/player/url"
        private val BR_MAP = mapOf(
            "flac" to 999000,
            "320k" to 320000,
            "192k" to 192000,
            "128k" to 128000,
        )
    }

    override suspend fun resolve(songmid: String, quality: Quality): PlayableUrl {
        val br = BR_MAP[quality.lxKey]
            ?: throw MusicSourceException.SourceDisabled("wy", "quality ${quality.lxKey} unsupported")
        val data = """{"ids":"[$songmid]","br":$br}"""
        val params = eapiEncrypt(EAPI_URL_PATH, data)

        val resp = http.post(ENDPOINT) {
            contentType(ContentType.Application.FormUrlEncoded)
            setBody("params=$params")
        }
        val text = resp.bodyAsText()
        val parsed = Json.parseToJsonElement(text).jsonObject
        val dataArr = parsed["data"]?.jsonArray
            ?: throw MusicSourceException.Parse("wy", null)
        val first = dataArr.firstOrNull()?.jsonObject
            ?: throw MusicSourceException.Parse("wy", null)
        val url = first["url"]?.jsonPrimitive?.contentOrNull
            ?: throw MusicSourceException.UrlExpired("wy")

        return PlayableUrl(
            url = url,
            quality = quality,
            fetchedAt = kotlin.time.Clock.System.now(),
        )
    }

    /** eapi 签名：AES-ECB(key="e82ckenh8dichen8") 加密 "nobody{url}use{data}md5forencrypt" 的 MD5 + 原 url + data */
    private fun eapiEncrypt(url: String, data: String): String {
        val message = "nobody${url}use${data}md5forencrypt"
        val digest = crypto.md5Hex(message.encodeToByteArray())
        val plain = "${url}-36cd479b6b5-${data}-36cd479b6b5-${digest}"
        val encrypted = crypto.aesEncrypt(
            plain.encodeToByteArray(),
            EAPI_KEY.encodeToByteArray(),
            iv = null,
            mode = "ECB",
        )
        return crypto.hexEncode(encrypted).uppercase()
    }
}
```

**⚠️ 方案切换记录（2026-04-20）：** 本 Task 原设计走 JS 引擎调 lx vendor `infSign.min.js`。vendor audit 发现：

- `infSign` 的实际签名是 `infSign(paramsObj, null, {useH5, isCDN, callback(i){return i.signature}})` callback 风格异步，在 vendor 里只用于 ranking/songList，不是 URL 拿取。
- 社区 mitmproxy 资料显示 trackercdn 端点其实可以用 `key = md5(hash.lowercase() + "kgcloudv2")` 简单签名拿到 URL，**不需要 infSign，不需要 JS 桥**。

- [ ] **Step 1: 实现 `KgUrlResolver.kt`**

```kotlin
package top.iwesley.lyn.music.online.resolve

import io.ktor.client.HttpClient
import io.ktor.client.request.*
import io.ktor.client.statement.*
import kotlinx.serialization.json.*
import top.iwesley.lyn.music.online.source.OnlineJson
import top.iwesley.lyn.music.online.source.PlatformCrypto
import top.iwesley.lyn.music.online.types.PlayableUrl
import top.iwesley.lyn.music.online.types.Quality
import top.iwesley.lyn.music.scripting.MusicSourceException

/**
 * kg 源（酷狗）URL 拿取器。
 *
 * lx kg vendor SDK 本身不实现 getMusicUrl（vendor/lx-sdk/kg/index.js:18-19 路由到
 * api-source 插件）。本实现使用 `trackercdn.kugou.com/i/v2/?cmd=26` 这条社区
 * mitmproxy 约定的端点；签名方案：`key = md5(hash.lowercase() + "kgcloudv2").hex_lower`，
 * 纯 Kotlin 实现，无需 JS 桥、无需 vendor infSign。
 *
 * M1.0 不按 quality 切换 cmd（不同 cmd 拿不同音质；hi-res 走 cmd=27）；BR_MAP 仅用于
 * gating 不支持的 quality。M1.1 接入 cmd 切换。
 *
 * 响应形如 `{"status":1,"url":["..."]}`；status != 1 或 url 数组空 → UrlExpired。
 */
class KgUrlResolver(
    private val http: HttpClient,
    private val crypto: PlatformCrypto,
) : SourceUrlResolver {

    companion object {
        private const val KEY_SALT = "kgcloudv2"
        private val BR_MAP = mapOf(
            "flac" to "flac",
            "320k" to "320",
            "192k" to "192",
            "128k" to "128",
        )
    }

    override suspend fun resolve(songmid: String, quality: Quality): PlayableUrl {
        BR_MAP[quality.lxKey]
            ?: throw MusicSourceException.SourceDisabled("kg", "quality ${quality.lxKey} unsupported")

        val hashUpper = songmid.uppercase()
        val key = kgKey(songmid)

        val resp = http.get("http://trackercdn.kugou.com/i/v2/") {
            url.parameters.append("cmd", "26")
            url.parameters.append("hash", hashUpper)
            url.parameters.append("key", key)
            url.parameters.append("pid", "6")
            url.parameters.append("behavior", "play")
            url.parameters.append("album_id", "0")
            url.parameters.append("album_audio_id", "0")
        }
        if (!resp.status.isSuccess()) {
            throw MusicSourceException.Network("kg", resp.status.value)
        }

        val text = resp.bodyAsText()
        val parsed = try {
            OnlineJson.parseToJsonElement(text).jsonObject
        } catch (e: Throwable) {
            throw MusicSourceException.Parse("kg", e)
        }

        val status = parsed["status"]?.jsonPrimitive?.intOrNull ?: parseErr("status")
        val urlArr = parsed["url"]?.jsonArray
        if (status != 1 || urlArr.isNullOrEmpty()) {
            throw MusicSourceException.UrlExpired("kg")
        }
        val playUrl = urlArr.first().jsonPrimitive.contentOrNull
            ?: throw MusicSourceException.UrlExpired("kg")

        return PlayableUrl(
            url = playUrl,
            quality = quality,
            fetchedAt = kotlin.time.Clock.System.now(),
        )
    }

    internal fun kgKey(hash: String): String =
        crypto.md5Hex((hash.lowercase() + KEY_SALT).encodeToByteArray())

    private fun parseErr(field: String): Nothing =
        throw MusicSourceException.Parse("kg", IllegalStateException("missing field: $field"))
}
```

```kotlin
import io.ktor.client.request.*

import top.iwesley.lyn.music.online.types.Quality

/**
 * mg 源（咪咕）URL 拿取器。
 * 端点：http://app.c.nf.migu.cn/MIGUM2.0/v1.0/content/resourceinfo.do?by=orpheus
 */
class MgUrlResolver(
    private val http: HttpClient,
    private val crypto: PlatformCrypto,
) : SourceUrlResolver {

    companion object {
        private val CODE_MAP = mapOf(
            "flac24bit" to "ZQ",
            "flac" to "SQ",
            "320k" to "HQ",
            "128k" to "PQ",
        )
    }

    override suspend fun resolve(songmid: String, quality: Quality): PlayableUrl {
        val code = CODE_MAP[quality.lxKey]
            ?: throw MusicSourceException.SourceDisabled("mg", "quality ${quality.lxKey} unsupported")

        val url = "http://app.c.nf.migu.cn/MIGUM2.0/v1.0/content/resourceinfo.do?resourceId=$songmid&resourceType=2&by=orpheus"
        val resp = http.get(url) {
            header("User-Agent", "Mozilla/5.0 (Linux; Android 11) MiguMusic/7.7.2")
        }
        val text = resp.bodyAsText()
        val parsed = Json.parseToJsonElement(text).jsonObject
        val resources = parsed["resource"]?.jsonArray
            ?: throw MusicSourceException.Parse("mg", null)
        val first = resources.firstOrNull()?.jsonObject
            ?: throw MusicSourceException.UrlExpired("mg")
        val rateformats = first["newRateFormats"]?.jsonArray ?: first["rateFormats"]?.jsonArray
            ?: throw MusicSourceException.Parse("mg", null)
        val match = rateformats.firstOrNull {
            it.jsonObject["formatType"]?.jsonPrimitive?.content == code
        }?.jsonObject ?: throw MusicSourceException.UrlExpired("mg")
        val playUrl = match["url"]?.jsonPrimitive?.contentOrNull
            ?: match["androidUrl"]?.jsonPrimitive?.contentOrNull
            ?: throw MusicSourceException.UrlExpired("mg")

        return PlayableUrl(
            url = if (playUrl.startsWith("//")) "http:$playUrl" else playUrl,
            quality = quality,
            fetchedAt = kotlin.time.Clock.System.now(),
        )
    }
}
```

## 风险快照

1. **Apple RSA SPKI 剥头定长假设**：`stripSpkiHeader` 只对 1024/2048-bit 精确。若 wy 真实 key 换 length，需扩算法
2. **CCCrypt 指针操作** 在 Kotlin/Native 的 pin/unpin 时序需谨慎，容易 SEGV；先跑 macosArm64Test 对比 JVM 输出
3. **platform.zlib.inflate** 在 K/N 2.3.20 的 cinterop 签名可能与 plan 代码略有差异（`z_stream` 字段命名、`Z_NO_FLUSH` 常量路径），按 IDE auto-import 调整
4. **QuickJS `__lyn_makePromise` 调用时序**：`ctx.evaluate("globalThis.__lyn_makePromise()")` 必须在 JSCallFunction 回调内部（同 JS 线程）调用，否则 `checkSameThread` 抛
5. **kg jsHelper wiring 遗留 TODO**：M1.0 kg URL 暂时抛 NotImplementedError；findMusic 跨源会用其他源补救；M1.1 或手动补 wiring
6. **tx zzcSign 简化版可能被拒**：`SHA1(payload+salt)` 是简化实现，真实 lx 版本是 MD5+SHA1 拼接。M1.0 真机冒烟若 tx 返回 `sign invalid` 需补真算法
7. **Apple register 仍是 runBlocking**：M1.0 Apple 不走 Promise 桥（推迟 M1.1 Task 4）；host 函数若 suspend IO 会阻塞 JSC 线程，但 M1.0 已切到 Kotlin Resolver 路径 for URL，JS 侧只跑 search/lyric/pic（bridge 同步 compute 为主 + request async）。request 在 Apple 侧靠 runBlocking(dispatcher) 同步等 Ktor response，可能有限定性能问题但不会死锁（Ktor engine 不会 withContext 回 JSC dispatcher）
